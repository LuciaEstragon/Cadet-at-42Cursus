# \033[48;2;R;G;Bm es el código ANSI para color de fondo RGB
def dibujar_cuadrado_rgb(r, g, b):
    color = f"\033[48;2;{r};{g};{b}m"
    reset = "\033[0m"
    # Imprimimos 2 espacios (que juntos forman un cuadrado en pantalla)
    print(f"{color}  {reset}", end="")

dibujar_cuadrado_rgb(255, 87, 51) # Color naranja
